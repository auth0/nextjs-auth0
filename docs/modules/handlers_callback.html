AfterCallback

AfterCallback: (req: NextApiRequest, res: NextApiResponse, session: default, state: {}) => Promise<default> | default
  • Defined in src/handlers/callback.ts:59
  • Defined in src/handlers/index.ts:1

Use this function for validating additional claims on the user's access token or adding removing items from the session after login, eg

Validate additional claims

 
// pages/api/auth/[...auth0].js
import { handleAuth, handleCallback } from '@auth0/nextjs-auth0';

const afterCallback = (req, res, session, state) => {
  if (!session.user.isAdmin) {
    throw new UnauthorizedError('User is not admin');
  }
  return session;
};

export handleAuth({
  async callback(req, res) {
    try {
      await handleCallback(req, res, { afterCallback });
    } catch (error) {
      res.status(error.status || 500).end(error.message);
    }
  }
});

Modify the session after login

 
// pages/api/auth/[...auth0].js
import { handleAuth, handleCallback } from '@auth0/nextjs-auth0';

const afterCallback = (req, res, session, state) => {
  session.user.customProperty = 'foo';
  delete session.refreshToken;
  return session;
};

export handleAuth({
  async callback(req, res) {
    try {
      await handleCallback(req, res, { afterCallback });
    } catch (error) {
      res.status(error.status || 500).end(error.message);
    }
  }
});
